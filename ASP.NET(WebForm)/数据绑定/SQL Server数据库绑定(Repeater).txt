				SqlCommand command=new SqlCommand("select * from titles where price!=0",connection);
				SqlDataReader reader=command.ExecuteReader();
				this.Repeater1.DataSource=reader;
				this.Repeater1.DataBind();
			}
			finally
			{
				connection.Close();
